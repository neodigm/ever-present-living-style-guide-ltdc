$(document).ready(function() {
  var owl = $("#owl-artc"), pImg = $("#caro_prev_img"), nImg = $("#caro_next_img");
  owl.owlCarousel({
      rewindNav: false,
      addClassActive: true,
      afterMove: owlMoved,
      afterUpdate: owlMoved,
      itemsDesktop: [1199,5],
      itemsDesktopSmall: [979,5],
      itemsTablet: [768,2],
      itemsTabletSmall: [650,2],
      itemsMobile: [479,2],
      pagination: false
  });
  $(".togCaro").click(function(e) {
    owlMoved();
    $("#secCaro").toggleClass( "hide" );
    $(".divCaro").toggleClass( "hide" );
    $("#caroAll").toggleClass("caroAll_alt");
    $("#respn_panl_reslt").toggleClass("respn_panl_reslt_alt");
  });  
  $(".next").click(function(){
    owl.trigger('owl.next');
  })
  $(".prev").click(function(){
    owl.trigger('owl.prev');
  })
  function owlMoved() { 
    var owl = $("#owl-artc").data('owlCarousel');
    if( $( "#item_1" ).parent().hasClass( "active" )){
      $(pImg).attr("src", $(pImg).attr("src").replace('_e','_d'));
      $(pImg).addClass('nav_arrw_d');
    }else{
      $(pImg).attr("src", $(pImg).attr("src").replace('_d','_e'));
      $(pImg).removeClass('nav_arrw_d');
    }
    if($( "#item_"+owl.itemsAmount ).parent().hasClass( "active" )){
      $(nImg).attr("src", $(nImg).attr("src").replace('_e','_d'));
      $(nImg).addClass('nav_arrw_d');
    }else{
      $(nImg).attr("src", $(nImg).attr("src").replace('_d','_e'));
      $(nImg).removeClass('nav_arrw_d');
    }
  }
});
